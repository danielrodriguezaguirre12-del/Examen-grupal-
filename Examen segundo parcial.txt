# ============================================================================
# 1. SUBSTITUTE BYTES - Sustitución de bytes usando la S-box
# ============================================================================

def substitute_bytes(matrix):
    """Aplica SubstituteBytes a la matriz 4x4"""
    rows = len(matrix)
    cols = len(matrix[0])
    result = [[0 for _ in range(cols)] for _ in range(rows)]
    
    for i in range(rows):
        for j in range(cols):
            value = matrix[i][j]
            if 0 <= value < 256:
                result[i][j] = SBOX[value]
            else:
                result[i][j] = value
    
    return result

def inverse_substitute_bytes(matrix):
    """Aplica Inverse SubstituteBytes a la matriz 4x4"""
    rows = len(matrix)
    cols = len(matrix[0])
    result = [[0 for _ in range(cols)] for _ in range(rows)]
    
    for i in range(rows):
        for j in range(cols):
            value = matrix[i][j]
            if 0 <= value < 256:
                result[i][j] = INVERSE_SBOX[value]
            else:
                result[i][j] = value
    
    return result

# ============================================================================
# 2. SHIFT ROWS - Desplazamiento circular de filas
# ============================================================================

def shift_rows(state):
    """
    Aplica ShiftRows a la matriz de estado:
    - Fila 0: sin cambios
    - Fila 1: desplazamiento circular izquierdo de 1 posición
    - Fila 2: desplazamiento circular izquierdo de 2 posiciones
    - Fila 3: desplazamiento circular izquierdo de 3 posiciones
    """
    result = [[0 for _ in range(4)] for _ in range(4)]
    
    result[0] = state[0][:]
    result[1] = state[1][1:] + state[1][:1]
    result[2] = state[2][2:] + state[2][:2]
    result[3] = state[3][3:] + state[3][:3]
    
    return result

def inverse_shift_rows(state):
    """
    Aplica Inverse ShiftRows (desplazamiento a la derecha):
    - Fila 0: sin cambios
    - Fila 1: desplazamiento circular derecho de 1 posición
    - Fila 2: desplazamiento circular derecho de 2 posiciones
    - Fila 3: desplazamiento circular derecho de 3 posiciones
    """
    result = [[0 for _ in range(4)] for _ in range(4)]
    
    result[0] = state[0][:]
    result[1] = state[1][-1:] + state[1][:-1]
    result[2] = state[2][-2:] + state[2][:-2]
    result[3] = state[3][-3:] + state[3][:-3]
    
    return result

# ============================================================================
# 3. MIX COLUMNS - Multiplicación en GF(2^8)
# ============================================================================

def gmul(a, b):
    """Multiplicación en GF(2^8) (Galois Field)"""
    if b == 1:
        return a
    tmp = (a << 1) & 0xff
    if b == 2:
        return tmp if a < 128 else tmp ^ 0x1b
    if b == 3:
        return gmul(a, 2) ^ a
    return 0

def mix_single_column(col):
    """Aplica MixColumns a una columna individual"""
    a, b, c, d = col[0], col[1], col[2], col[3]
    
    return [
        gmul(a, 2) ^ gmul(b, 3) ^ gmul(c, 1) ^ gmul(d, 1),
        gmul(a, 1) ^ gmul(b, 2) ^ gmul(c, 3) ^ gmul(d, 1),
        gmul(a, 1) ^ gmul(b, 1) ^ gmul(c, 2) ^ gmul(d, 3),
        gmul(a, 3) ^ gmul(b, 1) ^ gmul(c, 1) ^ gmul(d, 2)
    ]

def mix_columns(state):
    """Aplica MixColumns a todas las columnas"""
    result = [[0 for _ in range(4)] for _ in range(4)]
    
    for col in range(4):
        column = [state[row][col] for row in range(4)]
        mixed = mix_single_column(column)
        for row in range(4):
            result[row][col] = mixed[row]
    
    return result

def inverse_mix_columns(state):
    """Aplica Inverse MixColumns"""
    result = [[0 for _ in range(4)] for _ in range(4)]
    
    for col in range(4):
        a, b, c, d = state[0][col], state[1][col], state[2][col], state[3][col]
        
        result[0][col] = gmul(a, 14) ^ gmul(b, 11) ^ gmul(c, 13) ^ gmul(d, 9)
        result[1][col] = gmul(a, 9) ^ gmul(b, 14) ^ gmul(c, 11) ^ gmul(d, 13)
        result[2][col] = gmul(a, 13) ^ gmul(b, 9) ^ gmul(c, 14) ^ gmul(d, 11)
        result[3][col] = gmul(a, 11) ^ gmul(b, 13) ^ gmul(c, 9) ^ gmul(d, 14)
    
    return result

# ============================================================================
# 4. ADD ROUND KEY - XOR con la clave redonda
# ============================================================================

def add_round_key(state, round_key):
    """Aplica AddRoundKey (XOR entre estado y clave redonda)"""
    result = [[0 for _ in range(4)] for _ in range(4)]
    
    for i in range(4):
        for j in range(4):
            result[i][j] = state[i][j] ^ round_key[i][j]
    
    return result

# ============================================================================
# KEY EXPANSION - Expansión de claves
# ============================================================================

def key_expansion(key):
    """
    Expande la clave de 16 bytes a 44 palabras (176 bytes) para AES-128
    Genera 11 claves redondas (una inicial + 10 para las rondas)
    """
    rcon = [0x01, 0x02, 0x04, 0x08, 0x10, 0x20, 0x40, 0x80, 0x1b, 0x36]
    
    w = []
    
    # Primeras 4 palabras son la clave original
    for i in range(4):
        w.append([key[4*i], key[4*i+1], key[4*i+2], key[4*i+3]])
    
    # Generar las 40 palabras restantes
    for i in range(4, 44):
        temp = w[i-1][:]
        
        if i % 4 == 0:
            # RotWord
            temp = temp[1:] + temp[:1]
            # SubWord
            temp = [SBOX[b] for b in temp]
            # XOR con rcon
            temp[0] ^= rcon[(i // 4) - 1]
        
        w.append([w[i-4][j] ^ temp[j] for j in range(4)])
    
    return w

def words_to_state_key(words_indices):
    """Convierte 4 palabras en una matriz de estado 4x4"""
    state = [[0 for _ in range(4)] for _ in range(4)]
    for col in range(4):
        for row in range(4):
            state[row][col] = words_indices[col][row]
    return state

# UTILIDADES DE IMPRESIÓN

def print_matrix(matrix, title=""):
    """Imprime una matriz en formato hexadecimal"""
    if title:
        print(f"\n{title}")
        print("-" * 40)
    for row in matrix:
        print(" ".join(f"{val:02x}" for val in row))

def print_state(state, title=""):
    """Imprime el estado en formato de columnas"""
    if title:
        print(f"\n{title}")
        print("-" * 40)
    for row in state:
        print(" ".join(f"{val:02x}" for val in row))

# CIFRADO Y DESCIFRADO AES-128
# ============================================================================

def aes_encrypt(plaintext, key):
    """
    Cifra el texto plano usando AES-128
    plaintext: lista de 16 bytes
    key: lista de 16 bytes
    """
    
    # Convertir a matriz 4x4 (almacenamiento por columnas)
    state = [[plaintext[row + col*4] for col in range(4)] for row in range(4)]
    
    # Expansión de clave
    w = key_expansion(key)
    
    print("\n" + "="*60)
    print("ENCRIPTACIÓN AES-128")
    print("="*60)
    print_state(state, "Plaintext inicial:")
    
    # AddRoundKey inicial
    round_key = words_to_state_key(w[0:4])
    state = add_round_key(state, round_key)
    print_state(state, "Después de AddRoundKey inicial:")
    
    # Rondas principales (1-9)
    for round_num in range(1, 10):
        print(f"\n--- RONDA {round_num} ---")
        
        # SubstituteBytes
        state = substitute_bytes(state)
        print_state(state, "Después de SubstituteBytes:")
        
        # ShiftRows
        state = shift_rows(state)
        print_state(state, "Después de ShiftRows:")
        
        # MixColumns
        state = mix_columns(state)
        print_state(state, "Después de MixColumns:")
        
        # AddRoundKey
        round_key = words_to_state_key(w[round_num*4:(round_num+1)*4])
        state = add_round_key(state, round_key)
        print_state(state, f"Después de AddRoundKey (ronda {round_num}):")
    
    # Ronda final (10)
    print("\n--- RONDA FINAL (10) ---")
    state = substitute_bytes(state)
    print_state(state, "Después de SubstituteBytes:")
    
    state = shift_rows(state)
    print_state(state, "Después de ShiftRows:")
    
    round_key = words_to_state_key(w[40:44])
    state = add_round_key(state, round_key)
    print_state(state, "Ciphertext final (después de AddRoundKey):")
    
    # Convertir matriz 4x4 a lista de 16 bytes
    ciphertext = []
    for row in range(4):
        for col in range(4):
            ciphertext.append(state[row][col])
    
    return ciphertext

def aes_decrypt(ciphertext, key):
    """
    Descifra el texto cifrado usando AES-128
    ciphertext: lista de 16 bytes
    key: lista de 16 bytes
    """
    
    # Convertir a matriz 4x4
    state = [[ciphertext[row + col*4] for col in range(4)] for row in range(4)]
    
    # Expansión de clave
    w = key_expansion(key)
    
    print("\n" + "="*60)
    print("DESENCRIPTACIÓN AES-128")
    print("="*60)
    print_state(state, "Ciphertext inicial:")
    
    # AddRoundKey inicial
    round_key = words_to_state_key(w[40:44])
    state = add_round_key(state, round_key)
    print_state(state, "Después de AddRoundKey inicial:")
    
    # Rondas inversas (9-1)
    for round_num in range(9, 0, -1):
        print(f"\n--- RONDA {round_num} (Inversa) ---")
        
        # Inverse ShiftRows
        state = inverse_shift_rows(state)
        print_state(state, "Después de Inverse ShiftRows:")
        
        # Inverse SubstituteBytes
        state = inverse_substitute_bytes(state)
        print_state(state, "Después de Inverse SubstituteBytes:")
        
        # AddRoundKey
        round_key = words_to_state_key(w[round_num*4:(round_num+1)*4])
        state = add_round_key(state, round_key)
        print_state(state, f"Después de AddRoundKey (ronda {round_num}):")
        
        # Inverse MixColumns
        state = inverse_mix_columns(state)
        print_state(state, "Después de Inverse MixColumns:")
    
    # Ronda final
    print("\n--- RONDA FINAL (Inversa) ---")
    state = inverse_shift_rows(state)
    print_state(state, "Después de Inverse ShiftRows:")
    
    state = inverse_substitute_bytes(state)
    print_state(state, "Después de Inverse SubstituteBytes:")
    
    round_key = words_to_state_key(w[0:4])
    state = add_round_key(state, round_key)
    print_state(state, "Plaintext final (después de AddRoundKey):")
    
    # Convertir matriz 4x4 a lista de 16 bytes
    plaintext = []
    for row in range(4):
        for col in range(4):
            plaintext.append(state[row][col])
    
    return plaintext

def main():
    print("\n")
    print("="*60)
    print("AES-128 COMPLETO - ENCRIPTACIÓN Y DESENCRIPTACIÓN")
    print("="*60)
    
    # Plaintext de ejemplo: "Hello World 1234" (16 caracteres)
    plaintext = [ord(c) for c in "Hello World 1234"]
    
    # Clave de ejemplo: "MySecret123456!!" (16 caracteres)
    key = [ord(c) for c in "MySecret123456!!"]
    
    print("\nDatos de entrada:")
    print(f"Plaintext: {plaintext}")
    print(f"Plaintext (texto): {''.join(chr(b) for b in plaintext)}")
    print(f"Key: {key}")
    print(f"Key (texto): {''.join(chr(b) for b in key)}")
    
    # Encriptar
    ciphertext = aes_encrypt(plaintext, key)
    
    print("\n" + "="*60)
    print("CIPHERTEXT FINAL:")
    print("="*60)
    print(f"Bytes: {ciphertext}")
    print(f"Hexadecimal: {' '.join(f'{b:02x}' for b in ciphertext)}")
    
    # Desencriptar
    decrypted = aes_decrypt(ciphertext, key)
    
    print("\n" + "="*60)
    print("PLAINTEXT RECUPERADO:")
    print("="*60)
    print(f"Bytes: {decrypted}")
    print(f"Texto: {''.join(chr(b) for b in decrypted)}")
    
    # Verificar que coincide con el original
    if decrypted == plaintext:
        print("\n ÉXITO: El texto descifrado coincide con el original!")
    else:
        print("\n ERROR: El texto descifrado NO coincide con el original!")

if __name__ == "__main__":
    main()
